'use strict'
import {isBoolean, assert, Sequence} from 'flexio-jshelpers'
import {EventListenerParam} from './EventListenerParam'

const _isDispatching_ = Symbol('_isDispatching_')
const _sequenceId_ = Symbol('_sequenceId_')

/**
 * @class
 * @abstract
 */
export class EventHandlerBase {
  constructor() {
    /**
     *
     * @params {Map<any, any>}
     * @protected
     */
    this._listeners = new Map()
    this._pendingPayload = new Map()
    /**
     *
     * @params {Set<string>}
     * @protected
     */
    this._isHandled = new Set()
    /**
     *
     * @params {Set<string>}
     * @protected
     */
    this._isPending = new Set()
    this[_sequenceId_] = new Sequence(this.ID)

    /**
     * @property {boolean}
     * @name EventHandlerBase#Symbol(_isDispatching_)
     * @private
     */
    var _isDispatching = false
    Object.defineProperty(this, _isDispatching_, {
      enumerable: false,
      configurable: false,
      get: () => _isDispatching,
      set: (v) => {
        assert(isBoolean(v),
          'hotballoonView:dispatcher: `newIsDispatching` argument assert be a Boolean'
        )
        _isDispatching = v
      }
    })
  }

  /**
   *
   * @param {String} events
   * @param {Object} payload
   */
  dispatch(events, payload) {
    if (this._listeners.has(events)) {
      this._beforeDispatch(events, payload)
      try {
        this._listeners.get(events).forEach((v, k) => {
          if (!this._isPending.has(k)) {
            this._invokeCallback(events, k)
          }
        })
      } finally {
        this._stopDispatching(events)
      }
    }
  }

  /**
   *
   * @param {string} events
   * @protected
   */
  _ensureHaveListenersMap(events) {
    if (!(this._listeners.has(events))) {
      this._listeners.set(events, new Map())
    }
  }

  /**
   *
   * @private
   * @param {String} events of Listener
   * @param {String} token : token of listener
   */
  _invokeCallback(events, token) {
    this._isPending.add(token)
    try {
      this._listeners.get(events)
        .get(token)
        .callback(this._pendingPayload.get(events), events)
    } finally {
      this._isHandled.add(token)
    }
  }

  /**
   *
   * @return {string}
   */
  nextID() {
    return this[_sequenceId_].nextID()
  }

  /**
   *
   * @param {EventListenerParam} eventListenerParam
   * @returns {String} token
   */
  addEventListener(eventListenerParam) {
    assert(eventListenerParam instanceof EventListenerParam,
      'hotballoon:EventHandlerBase:addEventListener: ̀`eventListenerParam` argument assert be an instance of EventListenerParam'
    )

    this._ensureHaveListenersMap(eventListenerParam.events)
    const id = this.nextID()

    this._listeners.get(eventListenerParam.events)
      .set(id, {
        callback: eventListenerParam.callback
      })

    return id
  }

  /**
   *
   * @param {String} events of Listener
   * @param {String} token : token
   * @throws AssertionError
   */
  removeEventListener(events, token) {
    if (this._listeners.has(events)) {
      assert(this._listeners.get(events).has(token),
        'hotballoon:EventHandlerBase:removeEventListener: ̀`id` argument not in _listeners : `%s`',
        events
      )
      this._listeners.get(events).delete(token)
    }
  }

  /**
   *
   * @param {String} events of Listener
   * @param {String} token : token
   * @returns {boolean}
   */
  hasEventListener(events, token) {
    return (this._listeners.has(events)) && (this._listeners.get(events).has(token))
  }

  /**
   *
   * @param {String} events of Listener
   * @param {Object} payload
   * @private
   */
  _beforeDispatch(events, payload) {
    this._listeners.get(events).forEach((v, k) => {
      this._isHandled.delete(k)
    })
    this._pendingPayload.set(events, payload)
    this[_isDispatching_] = true
  }

  /**
   *
   * @param {string} events
   * @protected
   */
  _stopDispatching(events) {
    this._listeners.get(events).forEach((v, k) => {
      this._isPending.delete(k)
    })
    this._pendingPayload.delete(events)
    this[_isDispatching_] = false
  }

  /**
   *
   * @return {boolean}
   */
  isDispatching() {
    return this[_isDispatching_]
  }
}
